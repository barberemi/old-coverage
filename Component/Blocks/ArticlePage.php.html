601    protected function articleSidebarTopStoriesFilterSources(mixed $topStories, array $mostViewed, Article $article): array
602    {
603        // Give me the id of the articles in $mostViewed
604        $mostViewedIds = array_map(fn (Article $mostViewedArticle) => $mostViewedArticle->getId(), $mostViewed);
605        // Exclude from $topStories articles that are already in $mostViewed (to avoid duplicates)
606        $topStories = array_filter($topStories, fn (Article $topStory) =>
607            // Exclude the current article from the top stories and the articles that are already in $mostViewed
608            !in_array($topStory->getId(), $mostViewedIds) && $topStory->getId() !== $article->getId());
609        $articleSpecification = new ArticleTopStoriesSpecification();
610        // If $topStories is not full (it happens often on vertical own design articles), fill it with latest articles of the same vertical (excluding the current article and the articles that are already in $mostViewed)
611        if (!$articleSpecification->isSatisfiedBy($topStories)) {
612            // Give me the id of the articles in $topStories
613            $topStoriesId = array_map(fn (Article $topStory) => $topStory->getId(), $topStories);

615            $articleIdsToexclude = array_merge($mostViewedIds, [$article->getId()], $topStoriesId);

617            $latestArticles = $this->getArticleApi()->getLatestArticlesForVertical($article->getVertical(), $articleIdsToexclude);

619            $topStories = array_merge($topStories, $latestArticles);
620        }
621
